function createActionButtons(doc) { const actionsCell = document.createElement('td'); // Edit Button const editButton = document.createElement('button'); editButton.textContent = 'Edit'; editButton.classList.add('normal-btn'); editButton.title = 'Edit Report'; editButton.onclick = function () { sessionStorage.setItem('inspectionStructureNumber', doc.structure_number); sessionStorage.setItem('inspectionDate', doc.date); window.open('../inspection/inspection.html', '_blank'); }; actionsCell.appendChild(editButton); // Create BCI button const bciButton = document.createElement('button'); bciButton.textContent = 'BCI'; bciButton.classList.add('bci-button'); bciButton.addEventListener('click', async function () { console.log('BCI button clicked for date:', doc.date); try { // 1. Get structure info const structureId = sessionStorage.getItem('structureId'); const structureName = sessionStorage.getItem('structureName'); if (!structureId || !structureName) { throw new Error('Missing structure information'); } // 2. Fetch bridge data to get span count const bridgeResponse = await fetch(`http://localhost:3000/api/bridges/${structureId}`); if (!bridgeResponse.ok) throw new Error('Failed to fetch bridge data'); const bridge = await bridgeResponse.json(); const totalSpans = bridge.span_number || 1; // 3. Fetch ALL defects for this structure const defectsResponse = await fetch( `http://localhost:3000/api/defectsbci?structureId=${structureId}` ); if (!defectsResponse.ok) throw new Error('Failed to fetch defects'); const allDefects = await defectsResponse.json(); // 4. Filter defects by inspection date CLIENT-SIDE const defectsForThisDate = allDefects.filter(defect => { return defect.inspection_date === doc.date; }); // 5. Fetch elements const elementsResponse = await fetch('http://localhost:3000/api/elements'); if (!elementsResponse.ok) throw new Error('Failed to fetch elements'); const elements = await elementsResponse.json(); // 6. Create SINGLE window const bciWindow = window.open("", "BCI Report", "width=1200,height=800,scrollbars=yes"); if (!bciWindow) { throw new Error('Popup window was blocked. Please allow popups for this site.'); } // 7. Generate HTML header with enhanced styling let htmlContent = ` BCI Report - ${structureName}

BCI Report - ${structureName}

Inspection Date: ${doc.date}

`; // 8. Add content for each span (2 pages per span) for (let spanNum = 1; spanNum <= totalSpans; spanNum++) { // Filter data for this span const spanDefects = defectsForThisDate.filter(defect => defect.span == spanNum ); const spanElements = elements.filter(el => el.span_number == null || el.span_number == spanNum ); if (spanElements.length > 0 || spanDefects.length > 0) { const combinedData = combineData(spanElements, spanDefects, doc.date); const inspectorName = spanDefects[0]?.inspector_name || 'Not Recorded'; const bciCrit = spanDefects.length > 0 ? spanDefects[0].bci_crit : 'N/A'; const bciAv = spanDefects.length > 0 ? spanDefects[0].bci_av : 'N/A'; // ========== PAGE 1 (Original Content) ========== htmlContent += `
Span ${spanNum} - Page 1 of 2
Superficial General Principal Special Form
Inspector: ${inspectorName} Date: ${doc.date} Next inspection: Road Ref:
Bridge name: ${structureName} Bridge Ref: ${structureId} Bridge code
Primary deck form${bridge.primary_form || 'N/A'}
Map Ref: ${bridge.latitude?.toFixed(3) || 'N/A'}, ${bridge.longitude?.toFixed(3) || 'N/A'} OSE: ${bridge.OSE || 'N/A'} OSN: ${bridge.OSN || 'N/A'}
Primary deck material${bridge.primary_material || 'N/A'}
Span: ${spanNum} of ${totalSpans} Span Width (m): ${bridge.span || 'N/A'} Span Length (m): ${bridge.length || 'N/A'}
Secondary deck form${bridge.secondary_form || 'N/A'}
All above ground elements inspected Photograph
Secondary deck material${bridge.secondary_material || 'N/A'}
BCI crit: ${bciCrit} BCI ave: ${bciAv}
${generateBCITableRows(combinedData)}
No Description S Ex Def W P Cost Comments
Page 1 of 2 - Span ${spanNum}
`; // ========== PAGE 2 (Additional Details) ========== htmlContent += `
Span ${spanNum} - Page 2 of 2

Defect Summary

${generateDefectSummaryRows(combinedData)}
Element Defect Type Severity Extent Urgency

Recommended Actions

${generateActionRows(combinedData)}
Element Recommended Action Estimated Cost Priority

Visual Documentation

[Photo/Diagram of defects in Span ${spanNum}]

Inspector Notes

${generateNotesContent(spanDefects)}
BCI crit: ${bciCrit} BCI ave: ${bciAv}
Page 2 of 2 - Span ${spanNum}
`; } } // 9. Close HTML and write to window htmlContent += ``; bciWindow.document.open(); bciWindow.document.write(htmlContent); bciWindow.document.close(); bciWindow.focus(); } catch (error) { console.error('BCI generation failed:', error); alert(`Error: ${error.message}`); } }); // Append buttons to the actions cell actionsCell.appendChild(editButton); actionsCell.appendChild(bciButton); return actionsCell; } function combineData(elements, defects, selectedDate) { return elements.map(element => { const defect = defects.find(defect => defect.element_no === element.element_number && defect.inspection_date === selectedDate ); return { ...element, inspection_date: defect ? defect.inspection_date : '-', s: defect ? defect.s : '-', ex: defect ? defect.ex : '-', def: defect ? defect.def : '-', w: defect ? defect.w : '-', p: defect ? defect.p : '-', cost: defect ? defect.cost : '', comments_remarks: defect ? defect.comments_remarks : '', bci_crit: defect ? defect.bci_crit : '-', bci_av: defect ? defect.bci_av : '-' }; }); } function generateBCITableRows(data) { let rows = ""; data.forEach((item) => { rows += ` ${item.element_number} ${item.description} ${item.s} ${item.ex} ${item.def} ${item.w} ${item.p} ${item.cost} ${item.comments_remarks} `; }); return rows; } // Helper functions for generating additional content function generateDefectSummaryRows(data) { return data .filter(item => item.def !== '-') .map(item => ` ${item.element_number} ${item.description} ${getSeverityDescription(item.s)} ${getExtentDescription(item.ex)} ${getPriorityDescription(item.p)} `) .join('') || 'No significant defects recorded'; } function generateActionRows(data) { return data .filter(item => item.def !== '-' || item.comments_remarks) .map(item => ` ${item.element_number} ${item.comments_remarks || 'Monitoring recommended'} ${item.cost || '-'} ${getPriorityDescription(item.p)} `) .join('') || 'No specific actions required'; } function generateNotesContent(defects) { const uniqueNotes = [...new Set(defects.map(d => d.comments_remarks).filter(Boolean))]; return uniqueNotes.length > 0 ? uniqueNotes.map(note => `

• ${note}

`).join('') : '

No additional notes recorded

'; } function getSeverityDescription(code) { const map = { '1': 'Very Minor', '2': 'Minor', '3': 'Moderate', '4': 'Severe' }; return map[code] || code; } function getExtentDescription(code) { const map = { 'A': '<10%', 'B': '10-25%', 'C': '25-50%', 'D': '>50%' }; return map[code] || code; } function getPriorityDescription(code) { const map = { '1': 'Low', '2': 'Medium', '3': 'High', '4': 'Critical' }; return map[code] || code; }
